    - Import `buildZodSchema`, `flattenZodErrors` from `@/shared/utils/zod-schema-builder`
    - Replace the manual `validate()` function with Zod-based validation:
      ```typescript
      function validate(): boolean {
        const allFieldsList = [...(props.sharedFields ?? []), ...(props.action?.formFields ?? [])]
        const schema = buildZodSchema(allFieldsList)
        const result = schema.safeParse(formData.value)
        if (!result.success) {
          errors.value = flattenZodErrors(result.error)
          return false
        }
        errors.value = {}
        return true
      }
      ```
